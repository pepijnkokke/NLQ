  \ifthenelse{\boolean{showHidden}}
             {#1}
             {}}

  \ifthenelse{\boolean{marginNotes}}
             {\marginpar{#1}}
             {#1}}

  \ifthenelse{\boolean{showNotes}}
             {\marginNote{\textcolor{red}{\textbf{TODO:~}#1}}}
             {}}
